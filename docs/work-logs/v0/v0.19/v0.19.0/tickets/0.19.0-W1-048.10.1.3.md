#### 5. 虛擬碼骨架

```
class FileContentReader:
  constructor({ parser, showError, readerFactory, detectFormat }):
    if not parser:
      throw TypeError('FileContentReader: parser is required')
    this.parser         = parser
    this.showError      = showError ?? noop
    this.readerFactory  = readerFactory ?? (() => new globalThis.FileReader())
    this.detectFormat   = detectFormat  ?? defaultDetectByExtension

  # ===== Public API =====
  read(file):
    fileFormat = this.detectFormat(file) ?? 'json'   # fallback 預設 'json'
    return new Promise((resolve, reject) => {
      reader = this.readerFactory()
      reader.onload  = (e) => this.handleSuccess(e, resolve, reject, fileFormat)
      reader.onerror = ()  => this.handleError(reject)
      reader.readAsText(file, 'utf-8')
    })

  # ===== Event handlers =====
  handleSuccess(e, resolve, reject, fileFormat):
    try:
      result = this.parser.parse(e.target.result, fileFormat)
      resolve(result)
    catch error:
      this.showError('載入檔案失敗：' + error.message)
      reject(error)         # 透傳，含原 code/details

  handleError(reject):
    msg = '讀取檔案時發生錯誤'
    this.showError(msg)
    err = new Error(msg)
    err.code = UNKNOWN_ERROR
    err.details = { category: 'general' }
    reject(err)

  # ===== Fallback detectFormat（無 deps.detectFormat 時）=====
  defaultDetectByExtension(file):
    if not file: return null
    name = file.name.toLowerCase()
    type = (file.type or '').toLowerCase()
    if name endsWith '.csv' or type startsWith 'text/csv': return 'csv'
    if name endsWith '.json' or type == 'application/json': return 'json'
    return null
```
