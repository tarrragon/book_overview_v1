#### 3. 虛擬碼骨架

```
const DEFAULT_DEPS = {
  detectFormatVersion: require('src/export/format-version-detector').detectFormatVersion,
  convertV1ToV2Data:   require('src/export/v1-to-v2-converter').convertV1ToV2Data
}

class ContentParser:
  constructor(deps = {}):
    this.detectFormatVersion = deps.detectFormatVersion ?? DEFAULT_DEPS.detectFormatVersion
    this.convertV1ToV2Data   = deps.convertV1ToV2Data   ?? DEFAULT_DEPS.convertV1ToV2Data
    this.largeDatasetThreshold = deps.largeDatasetThreshold ?? 1000

  # ===== Public API =====
  parse(content, fileFormat):
    # 前置鏈
    if fileFormat is undefined or null:
      throw TypeError('parse: fileFormat is required')
    if fileFormat not in ['json', 'csv']:
      throw makeError('parse: fileFormat must be "json" or "csv"', PARSE_ERROR, 'parsing')

    cleanContent = this.validateAndClean(content)    # VALIDATION_ERROR if empty
    data = (fileFormat == 'csv')
             ? this.parseCSV(cleanContent)
             : this.parseJSON(cleanContent)
    return this.processBookData(data, fileFormat)    # 含 INV-1 組裝

  # ===== Content cleaning =====
  validateAndClean(content):
    if not content or content.trim() == '':
      throw makeError('檔案內容為空', VALIDATION_ERROR, 'validation')
    return removeBOM(content)

  # ===== JSON path =====
  parseJSON(content):
    try: return JSON.parse(content)
    catch SyntaxError as e:
      throw makeError('JSON 檔案格式不正確', PARSE_ERROR, 'parsing', cause=e)

  # ===== CSV path（state machine 不變動，等價搬遷）=====
  parseCSV(content):
    rows = parseCSVRows(content)         # state machine 兩 helper
    if rows.length == 0:
      throw makeError(INVALID_CSV+'：內容為空', PARSE_ERROR, 'parsing')
    fieldNames = mapHeader(rows[0])
    if 'id' not in fieldNames or 'title' not in fieldNames:
      throw makeError(INVALID_CSV+'：缺少必要欄位', PARSE_ERROR, 'parsing')
    books = rows[1:].filter(not isBlankRow).map(csvRowToBook)
    return books

  # ===== Book data processing（六線性分支提取 + 過濾 + INV-1）=====
  processBookData(data, fileFormat):
    extracted = this.extractBooksFromData(data, fileFormat)
    validBooks = extracted.books.filter(isValidBook)
    this.checkLargeDataset(validBooks)
    return {
      books:         validBooks,
      tagCategories: extracted.tagCategories,
      tags:          extracted.tags
    }
```
