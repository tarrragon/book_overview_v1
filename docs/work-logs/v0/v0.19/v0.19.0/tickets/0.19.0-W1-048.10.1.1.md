#### 4. 虛擬碼骨架

```
class FileValidator:
  constructor(deps = {}):
    this.showError    = deps.showError ?? noop
    this.maxFileSize  = deps.maxFileSize ?? 10485760  # 10MB

  validate(file):
    # 前置鏈短路：file 存在 → 格式合法 → 大小合法
    if not file:
      this.showError('請先選擇一個 JSON 或 CSV 檔案！')
      throw makeError('檔案不存在', VALIDATION_ERROR, 'validation')

    format = this.detectFormat(file)
    if format is null:
      this.showError('請選擇 JSON 或 CSV 格式的檔案！')
      throw makeError('檔案格式不正確', VALIDATION_ERROR, 'validation')

    if file.size > this.maxFileSize:
      this.showError('檔案過大，請選擇小於 10MB 的檔案！')
      throw makeError('檔案大小超出限制', VALIDATION_ERROR, 'validation')

  detectFormat(file):
    if file is null or file is undefined:
      return null
    name = file.name.toLowerCase()
    type = (file.type or '').toLowerCase()
    if name endsWith '.json' or type == 'application/json':
      return 'json'
    if name endsWith '.csv' or type startsWith 'text/csv':
      return 'csv'
    return null

  # internal helper
  makeError(msg, code, category):
    err = new Error(msg)
    err.code = code
    err.details = { category }
    return err
```
