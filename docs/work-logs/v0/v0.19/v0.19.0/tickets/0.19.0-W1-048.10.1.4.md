    validate: jest.fn(),
    detectFormat: jest.fn(() => 'json'),
    ...overrides
  }
}
function createStubReader (overrides = {}) {
  return {
    read: jest.fn().mockResolvedValue({ books: [], tagCategories: [], tags: [] }),
    ...overrides

  return {
    parse: jest.fn().mockReturnValue({ books: [], tagCategories: [], tags: [] }),

  return {
    validator: createStubValidator(overrides.validator),
    reader: createStubReader(overrides.reader),

}

#### A. BookFileImporter 改寫策略

**A.1 Constructor DI 注入點**

當前（src/overview/book-file-importer.js L67）：

```
constructor ({ document, showError }) {
  this.document = document
  this.showError = showError
  // ...無 helper field
}
```

改寫為（依父 ticket Solution §B）：

```
constructor ({ document, showError, validator?, reader?, parser? }) {
  this.document = document
  this.showError = showError

  // 順序關鍵：parser 先於 reader（reader 預設依賴 parser）
  this.parser    = parser    ?? new ContentParser()
  this.validator = validator ?? new FileValidator({ showError })
  this.reader    = reader    ?? new FileContentReader({
                                   parser: this.parser,
                                   showError,
                                   detectFormat: (file) => this.validator.detectFormat(file)
                                 })
}
```

**設計要點**：
- 三 helper 全選用：production 路徑零變更（自建預設實例），test 路徑可注入 stub
- reader 預設 detectFormat 走 validator 委派，保持 validator 為 format 偵測 SSOT
- import 順序：檔頭 require './import/file-validator' / './import/file-reader' / './import/content-parser'（待 10.1.1~10.1.3 落地後存在）

**A.2 4 個 public method 改 thin wrapper**

| Method | 當前實作（行號） | 改寫後虛擬碼 |
|--------|----------------|------------|
| `handleFileLoad(file)` | L82, 內呼 `validate` + `read` | `this.validate(file); return await this.read(file)`（行為等價） |
| `validate(file)` | L97, 內呼 `_validateFileBasics` + `_validateFileSize` | `return this.validator.validate(file)` |
| `read(file)` | L111, 內呼 `_readFileWithReader` | `return this.reader.read(file)` |
| `parseContent(content, fileFormat)` | L127, 含 fileFormat 缺漏 / 非 'json'\|'csv' TypeError + `_handleFileContent` | `return this.parser.parse(content, fileFormat)`（TypeError/PARSE_ERROR 檢查移入 parser.parse 內部，10.1.2 已實作） |

**Wrapper 厚度約束**：每個 wrapper ≤ 3 行（無 try/catch / 無條件分支），保持「純 delegate」純度。errorshowError 副作用由各 helper 內部處理（既有設計）。
