```python
def run_dispatch_check(args) -> int:
    dispatch_file = Path(".claude/dispatch-active.json")
    if not dispatch_file.exists():
        print("[PASS] 無活躍派發，可繼續")
        return 0
    try:
        data = json.loads(dispatch_file.read_text(encoding="utf-8"))
    except json.JSONDecodeError:
        print("[FAIL] dispatch-active.json 格式錯誤", file=sys.stderr)
        return 2  # IO error（參考 handoff_ready IO_ERRORS 保守 NO-GO）
    dispatches = data.get("dispatches", [])
    if not dispatches:
        print("[PASS] 無活躍派發，可繼續")
        return 0
    print(f"[WARN] 有 {len(dispatches)} 個活躍派發：")
    for d in dispatches:
        desc = d.get("agent_description", "(unknown)")
        tid = d.get("ticket_id", "") or "(no ticket)"
        ts = d.get("dispatched_at", "")
        print(f"  - {desc} | ticket: {tid} | {ts}")
    return 1
```
