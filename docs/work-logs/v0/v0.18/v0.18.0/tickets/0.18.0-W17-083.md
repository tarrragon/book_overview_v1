HANDOFF_KEYWORDS = (
    "下個 Session 接手 Context", "下一 Session 接手", "下 Session 接手",
    "接手指引", "Handoff Context", "Session Handoff",
    "下 session 優先建議", "下個 session 優先建議", "下一 session 優先建議",
    "下 session 優先順序建議", "下次 session 建議", "建議下 session",
    "未完成清單", "Spawned 推進清單",
)
# 預編譯 regex 為模組級 const

def detect_handoff_keywords(content):
    return any(kw in content for kw in HANDOFF_KEYWORDS)

def extract_ticket_ids(content, active_version=None):
    seen = []
    seen_set = set()
    # 1) full pattern 先掃（避免 short pattern 重複命中）
    for m in TICKET_ID_FULL_PATTERN.finditer(content):
        tid = f"{m.group(1)}-{m.group(2)}"
        if tid not in seen_set:
            seen.append(tid); seen_set.add(tid)
    # 2) short pattern：補全 active_version
    if active_version:
        for m in TICKET_ID_SHORT_PATTERN.finditer(content):
            short = m.group(1)
            tid = f"{active_version}-{short}"
            if tid not in seen_set:
                seen.append(tid); seen_set.add(tid)
    return seen

def extract_recent_content(worklog_path, since_mtime):
    if not worklog_path.exists():
        return ""
    if worklog_path.stat().st_mtime < since_mtime:
        return ""
    return worklog_path.read_text(encoding="utf-8")
```

    worklog_path = args.worklog_path or _auto_detect_worklog()

            ticket_id=tid, version=None,
            to_parent=False, to_child=None, to_sibling=None,
            context_refresh=False, status=False, gc=False,
            dry_run=False, force=False, ...  # 補齊 _execute_handoff 預期欄位

def main():

### 3. 偵測流程虛擬碼

```
# Stop Hook 主邏輯（方案 A）

def on_session_stop(event):
    # 步驟 1: 找 active version 的 worklog
    active_version = read_active_version_from_todolist()
    worklog_path = find_worklog_path(active_version)
    if not worklog_path.exists():
        return  # 無 worklog 無需檢查

    # 步驟 2: 判斷 worklog 是否在本 session 內被修改
    session_start_time = event.get("session_start_timestamp")
    if worklog_path.mtime < session_start_time:
        return  # 本 session 未動 worklog

    # 步驟 3: 擷取 worklog 最新段落（末 N 行或本 session diff）
    recent_content = extract_recent_content(worklog_path, session_start_time)

    # 步驟 4: 關鍵字偵測
    if not any(kw in recent_content for kw in HANDOFF_KEYWORDS):
        return  # 無交接意圖

    # 步驟 5: 提取 ticket ID 清單
    worklog_ticket_ids = extract_ticket_ids(recent_content)  # regex match

    # 步驟 6: 比對 .claude/handoff/pending/ 現況
    pending_dir = Path(".claude/handoff/pending")
    pending_ticket_ids = {p.stem for p in pending_dir.glob("*.json")}

    missing = set(worklog_ticket_ids) - pending_ticket_ids
    orphan = pending_ticket_ids - set(worklog_ticket_ids)  # CLI 有但 worklog 無

    # 步驟 7: 輸出（四情境）
    if not missing and not orphan:
        return  # 雙軌一致，無需輸出
    output_warning(missing, orphan, suggested_commands)
```

    worklog_path = args.worklog_path or auto_detect_worklog()
    content = worklog_path.read_text()
    ticket_ids = extract_ticket_ids_from_handoff_section(content)

    for tid in ticket_ids:
        if (pending_dir / f"{tid}.json").exists():
            print(f"[SKIP] {tid}: 已存在 pending handoff")
            continue
        if args.dry_run:
            print(f"[DRY-RUN] 將執行：ticket handoff {tid}")
        else:
            # 呼叫既有 _execute_handoff 邏輯（或 subprocess ticket handoff <tid>）
            result = execute_handoff(tid)
            print(f"[{'OK' if result == 0 else 'FAIL'}] {tid}")
```

### 8. 風險與邊界聲明

| 風險 | 緩解 |
|------|------|

| Session 無 worklog 變更但有 CLI handoff | 屬情境 3「孤立」，輸出低優先級提醒，不阻塞 |

### 9. 推薦方案總結

**方案 D：組合 A（Stop Hook 偵測）+ C（CLI --from-worklog 子命令）**

**理由**：
1. Stop hook 自動偵測解決「PM 忘記執行 CLI」的核心漏報問題
2. CLI 子命令提供「被提醒後一鍵落地」的最低摩擦執行路徑
3. 兩者共用 `worklog_parser.py` 避免 DRY 違反
4. 與既有紀律規則（session-switching-sop.md v1.1.0）形成「紀律 + 自動化」雙層防護
5. 總實作成本約 3.5-4 小時，ROI 最高
